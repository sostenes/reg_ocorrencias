// Instanciando o objeto Date()

diasDaSemana[1] = "Segunda-feira";

// Referenciando aos meses do ano através de Array
var mes = new Array();

mes[1] = "Fevereiro";
mes[2] = "Março";
mes[3] = "Abril";
mes[4] = "Maio";
mes[5] = "Junho";
mes[6] = "Julho";
mes[7] = "Agosto";
mes[8] = "Setembro";
